import math


def fact(n):
    result = 1
    while n > 1:
        result *= n
        n -= 1
    return result

# 1.перестановка +
def f1(n, k):
    result = fact(n)
    return result

# 2.Сочетание +
def f2(n, k):
    result = int(fact(n) // (fact(k) * fact(n - k)))
    return result

# 3.Размещение +

    result = int(fact(n) // (fact(n - k)))
